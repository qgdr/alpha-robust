\section{Introduction}
We study $\gamma\in (0,1)$, $\alpha\in (1,2)$ and $\Omega=(0,2L)$.
\begin{equation}
  D_t^\gamma u + (-\Delta)^{\frac{\alpha}{2}} u = f(x,t), \quad x \in \Omega, t\in (0,T].
\end{equation}
where
\begin{equation}
  \begin{aligned}
    D_t^\gamma u(x, t) = \frac{1}{\Gamma(1-\gamma)} \int_0^t \frac{\partial u(x, s)}{\partial s} (t-s)^{-\gamma} ds
  \end{aligned}
\end{equation}
\begin{equation}
  \begin{aligned}
    (-\Delta)^{\frac{\alpha}{2}} u(x,t) = \frac{1}{2\cos(\alpha\pi/2) \Gamma(2-\alpha)} \int_{0}^{2L} u(y, t) |x-y|^{1-\alpha} dy
  \end{aligned}
\end{equation}
where $\gamma \in (0,1)$, $\alpha \in (1,2)$.

\section{Regularity of the solution}
For the space-time fractional diffusion equation, it was first assumed that the solution regularity
satisfies \cite{ROSOTON2014275,Stynes:2017,Xavier:2016,Harnack:2017}
\begin{subequations} \label{eq:regularity}
  \begin{equation} \label{subeq:regularity-on-t}
    \left|\frac{\partial^{l} u}{\partial t^l}(x,t)\right| \le C(1+t^{\gamma-l}) \quad for \quad l = 0,1,2,
  \end{equation}
  \begin{equation} \label{subeq:regularity-on-x}
    \left|\frac{\partial^{l} u}{\partial x^l}(x,t)\right| \le C\delta(x)^{\alpha/2-l} \quad for \quad l = 0,1,2,3,4,
  \end{equation}
  \begin{equation} \label{subeq:regularity-of-Au-on-x}
    \left|\frac{\partial^{l} }{\partial x^l} (-\Delta)^{\alpha/2} u(x,t) \right| \le C \delta(x)^{-\alpha/2-l} \quad for \quad l = 0,1,2,
  \end{equation}
\end{subequations}
for all $(x,t) \in (0, 2L)\times (0,T]$.
\begin{remark}
  \eqref{subeq:regularity-of-Au-on-x} can be derived from \eqref{subeq:regularity-on-x} by
  \begin{equation*}
    \begin{aligned}
      I^{2-\alpha} u(x,t) &= \int_{0}^{x/2} + \int_{L+x/2}^{2L} u(y,t) \frac{|x-y|^{1-\alpha}}{\Gamma(2-\alpha)} dy \\
      &\quad + \int_{0}^{x/2} \left(u(x-z,t) + u(x+z,t) \right) \frac{z^{1-\alpha}}{\Gamma(2-\alpha)} dy \\
      &\quad + \int_{x+x/2}^{L+x/2} u(y,t) \frac{|y-x|^{1-\alpha}}{\Gamma(2-\alpha)} dy
    \end{aligned}
  \end{equation*}
\end{remark}

\section{Numerical scheme}
\subsection{Discretisation of $(-\Delta)^{\frac{\alpha}{2}}$ on Graded Mesh}
We discretize the $(-\Delta)^{\alpha/2}$ on a graded mesh.
\begin{equation}
  -D_M^{\alpha}u(x_m, t_n) = -D_M^2 I^{2-\alpha} \Pi_M u (x_m, t_n)
\end{equation}
The discrete operator can be written in matrix form
% \begin{equation} \label{eq:equation_matrix}
%   A\hat{U}^n
% \end{equation}
\begin{equation}
  A = H^{-1} D, \quad \text{with} \quad
% \end{equation}
% where
% \begin{equation}
  H = \text{diag}\left( \frac{h_i+h_{i+1}}{2} \right)
% \end{equation}
~ and ~ 
% \begin{equation}
  D_{ij} = \frac{\kappa_\alpha}{\Gamma(4-\alpha)} C_i K_{ij} C_j^T
\end{equation}
$C_j := \left( \frac{1}{h_j}, -\frac{1}{h_j}-\frac{1}{h_{j+1}}, \frac{1}{h_{j+1}} \right)$
and
\begin{equation*}
    K_{ij} := \begin{pmatrix}
        |x_{i-1} - x_{j-1}|^{3-\alpha} & |x_{i-1} - x_{j}|^{3-\alpha} &|x_{i-1} - x_{j+1}|^{3-\alpha} \\
        |x_{i} - x_{j-1}|^{3-\alpha} & |x_{i} - x_{j}|^{3-\alpha} &|x_{i} - x_{j+1}|^{3-\alpha} \\
        |x_{i+1} - x_{j-1}|^{3-\alpha} & |x_{i+1} - x_{j}|^{3-\alpha} &|x_{i+1} - x_{j+1}|^{3-\alpha}
    \end{pmatrix} .
\end{equation*}
\begin{lemma}
  The stiffness matrix $A$ has the following properties:
  \begin{enumerate}
    \item The eigenvalues of $A=H^{-1} D$ are positive real numbers.
    Inparticular, $D$ is symmetric positive definite.
    % \item $A$ is positive definite, which means that the eigenvalues of $\frac{A+A^T}{2}$ are positive.
    \item The eigenvectors of $A$ are orthogonal in space where $\langle u,v \rangle := uHv$, 
    where $H := \text{diag}\left( \frac{h_i+h_{i+1}}{2} \right)$.
    \item $(I+\tau A)^{-1} > O$ for any $\tau > 0$, and $A^{-1}>O$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  Since
  \begin{equation}
    A = H^{-1} D = H^{-1/2} H^{-1/2} D H^{-1/2} H^{1/2} ,
  \end{equation}
  where $H^{-1/2} D H^{-1/2}$ is symmetric positive definite, $H^{-1/2} D H^{-1/2} = U\Lambda U^T$.
  Thus,
  \begin{equation}
    A = H^{-1/2} U \Lambda U^T H^{1/2} = (H^{-1/2} U) \Lambda (H^{-1/2} U)^{-1}.
  \end{equation}
  
  The eigenvectors of $A$ form an orthogonal basis of the Hilbert space defined by $\langle u,v \rangle := uHv$. 
  Let $v_i = H^{-1/2} u_i$ be an eigenvector of $A$ with eigenvalue $\lambda_i$.
\end{proof}

\begin{equation}

We have
\begin{equation}
  \begin{aligned}
    \left(d_{n,n} I + A \right) E^n = \sum_{k=1}^{n-1} (d_{n, k+1} - d_{n,k}) E^{k} + d_{n,1} E^0 + R^{n}
  \end{aligned}
\end{equation}



Define the matrices $\Theta_{n,j}$, for $n = 1, 2 . . . , N$ and $j = 0, 1, 2, . . . , n - 1$ by
\begin{equation} \label{eq:def_Theta}
  \begin{gathered}
    \Theta_{n,n} = (d_{n,n} + A)^{-1} , \quad \Theta_{0,0} = I, \quad
    \Theta_{n,j} = \sum_{k=j}^{n-1} (d_{n, k+1} - d_{n,k})\Theta_{n,n} \Theta_{k,j}. %\quad 1\le j <n 
    % \Theta_{n,0} = d_{n,1} (d_{n,n} + A)^{-1} + \sum_{k=1}^{n-1} (d_{n, k+1} - d_{n,k})(d_{n,n} + A)^{-1} \Theta_{k,0}
  \end{gathered}
\end{equation}
Observe that $\Theta_{n,j} > O$ for all $n, j$.

\begin{lemma}
  \begin{equation}
    \begin{aligned}
      E^n &= \sum_{j=1}^{n} \Theta_{n,j} R^{j} + \Theta_{n,0} E^{0} \\
      &= \sum_{j=1}^{n} \Theta_{n,j} R_t^{j} + \sum_{j=1}^{n} \Theta_{n,j} R_x^{j} + \Theta_{n,0} E^{0}
    \end{aligned}
  \end{equation}
\end{lemma}

Our stability result will be presented in a general framework. Assume that
\begin{equation} \label{eq:discrete_problem}
  \mathcal{L}_{M,N} v_m^n = g_m^n \quad \text{for} \quad 1\le m\le 2M-1, \quad 1\le n \le N,
\end{equation}
with $v_0^n = v_{2M}^n = 0$ for $0\le n\le N$ and $v_m^0$ given for $0\le m\le 2M$.

Here the discrete operator is $\mathcal{L}_{M,N} v_m^n = D_N^\gamma v_m^n - D_M^\alpha v_m^n$.

\begin{lemma} \label{lmm:stability}
  The solution of the discrete problem \eqref{eq:discrete_problem} satisﬁes
  \begin{equation*}
    \|v^n\|_\infty \le d_{n,n}^{-1} \left[ \|g^n\|_\infty + d_{n,1}\|v^0\|_\infty + \sum_{k=1}^{n-1} \left(d_{n, k+1} - d_{n,k}\right) \|v^{k}\|_\infty \right]
  \end{equation*}
  for $n=1,2,..., N$.
\end{lemma}
\begin{proof}
  Fix $n\in \{1,2,..., N\}$. Choose $i_0$ such that $|v^n_{i_0}| = \|v^n\|_\infty$. Then, it yields
  \begin{equation*}
    d_{n,n} v_{i_0} + \sum_{j=1}^{2M-1} a_{i_0, j} v^n_{j} = g^n_{i_0} + \sum_{k=1}^{n-1} \left(d_{n, k+1} - d_{n,k}\right) v^k_{i_0} + d_{n,1} v^0_{i_0}.
  \end{equation*}
  Hence, by $a_{i_0, i_0} > 0$ and the choice of $i_0$, one obtains
  \begin{equation*}
    \begin{aligned}
      \left(d_{n,n} + a_{i_0, i_0} \right) |v^n_{i_0}|
      &\le \sum_{j\neq i_0} |a_{i_0, j} | |v^n_j| + |g^n_{i_0}| + \sum_{k=1}^{n-1} \left(d_{n, k+1} - d_{n,k}\right) |v^k_{i_0}| + d_{n,1} |v^0_{i_0}|.
    \end{aligned}
  \end{equation*}
  Since $A$ is strcitly diagonally dominant, we can get
  \begin{equation*}
    d_{n,n} \|v^n\|_\infty \le \|g\|_\infty + d_{n,1}\|v^0\|_\infty + \sum_{k=1}^{n-1} \left(d_{n, k+1} - d_{n,k}\right) \|v^{k}\|_\infty.
  \end{equation*}
  The proof is completed.
\end{proof}



\begin{theorem} \label{thm:convergence-t}
  \begin{equation}
    \left|\sum_{j=1}^{n} \Theta_{n,j} R_t^{j}\right| \le C T^{\gamma} N^{-\min\{2-\gamma, r_t\gamma\}}.
  \end{equation}
\end{theorem}
\begin{proof}
  Let $v^n = \sum_{j=1}^{n} \Theta_{n,j} R_t^{j}$. Then, we can check that $v_n$ satisfies \eqref{eq:discrete_problem}
  \begin{equation*}
    \mathcal{L}_{M,N} \; v^n = R_t^{n}.
  \end{equation*}
  According to \Cref{thm:LTE_t,lmm:stability} and the proof of \cite[Theorem 5.3]{Stynes:2017}, one has 
  \begin{equation*}
    \|v^n\|_\infty \le C T^\gamma N^{-\min\{2-\gamma, r_t\gamma\}}.
  \end{equation*}
\end{proof}




\begin{lemma}
  For $n=1,2,..., N$, one has
  \begin{equation} \label{eq:sum_Theta}
    \sum_{j=1}^n \Theta_{n,j} < A^{-1}
    % \sum_{j=1}^n j^{-\beta} \Theta_{n,j} \le A^{-1}
  \end{equation}
\end{lemma}
\begin{proof}
  Use induction on $n$. When $n=1$, then
  $
    \sum_{j=1}^1  \Theta_{1,j} = \Theta_{1,1} < A^{-1}
  $.
  Next, assume that \eqref{eq:sum_Theta} holds for $k=1, 2,..., m-1(2\le m \le N)$. 
  We want to prove \eqref{eq:sum_Theta} for $n=m$.
  Invoking \eqref{eq:def_Theta} and interchanging the order of summation,
  \begin{equation*}
    \begin{aligned}
      \sum_{j=1}^m  \Theta_{m,j}
      &=  \Theta_{m,m} + \sum_{j=1}^{m-1}  \sum_{k=j}^{m-1} (d_{m, k+1} - d_{m,k})\Theta_{m,m} \Theta_{k,j} \\
      &= \Theta_{m,m} + \sum_{k=1}^{m-1} (d_{m, k+1} - d_{m,k})\Theta_{m,m}  \sum_{j=1}^{k} \Theta_{k,j} \\
      & \le  \Theta_{m,m} + \sum_{k=1}^{m-1} (d_{m, k+1} - d_{m,k})\Theta_{m,m}   A^{-1} \\
      &= \Theta_{m,m} + (d_{m, m} - d_{m,1})\Theta_{m,m}   A^{-1} \\
      &= A^{-1} - d_{m,1} \Theta_{m,m}   A^{-1} < A^{-1}
    \end{aligned}
  \end{equation*}
  The proof is completed.
\end{proof}

\begin{theorem} \label{thm:convergence-x}
  \begin{equation}
    \begin{aligned}
      \left| \sum_{j=1}^{n} \Theta_{n,j} R_x^{j} \right| \le C M^{-\min\{r_x\alpha/2, 2\}}
    \end{aligned}
  \end{equation}
\end{theorem}
\begin{proof}
  Since $\Theta_{n,j} > O$, we have
  \begin{equation}
    \begin{aligned}
      \left| \sum_{j=1}^{n} \Theta_{n,j} R_x^{j} \right| 
      &\le \sum_{j=1}^{n} \Theta_{n,j} |R_x^{j}|
      \le \sum_{j=1}^{n} \Theta_{n,j} |R_x|
      < A^{-1} R_x
    \end{aligned}
  \end{equation}
  Since $A^{-1} R_x$ is bounded by $C M^{-\min\{r_x\alpha/2, 2\}}$, the proof is completed.
\end{proof}

\begin{theorem}
  \begin{equation}
    |E_N| \le C \left( N^{-\min\{2-\gamma, r_t\gamma\}} + M^{-\min\{r_x\frac{\alpha}{2}, 2\}} \right)
  \end{equation}
\end{theorem}
\begin{proof}
  According to \Cref{thm:convergence-t,thm:convergence-x}, the desired result is obtained.
\end{proof}




\section*{One-order}
\begin{equation}
  \frac{\partial u}{\partial t} +  (-\Delta)^{\frac{\alpha}{2}} u = f(x,t), \quad x \in \Omega, t\in (0,T].
\end{equation}
scheme: Let $\tau = \frac{T}{M}$, $U^{n}, F^{n} \in \mathbb{R}^{2N-1}$,
\begin{equation}
  \frac{U^{n+1} - U^{n} }{\tau} + AU^{n+1} = F^{n+1}.
\end{equation}
Then $E^{n} = U^{n} - \hat{U}^n \in \mathbb{R}^{2N-1}$,
\begin{equation}
  (I+\tau A)E^{n+1} = E^{n} + \tau R^{n+1}.
\end{equation}
\begin{equation}
  \begin{aligned}
    E^{n} &= (I+\tau A)^{-1}E^{n-1} + (I+\tau A)^{-1}\tau R^{n} \\
    &= (I+\tau A)^{-n}E^{0} + \sum_{k=1}^{n} (I+\tau A)^{-k} \tau R^{n-k+1}
  \end{aligned}
\end{equation}
\begin{equation}
  \begin{aligned}
    (I+\tau A)^{-k} \tau R^{n-k+1} &= (\tau A) (I+\tau A)^{-k} (\tau A)^{-1} \tau R^{n-k+1} \\
    &= (\tau A) (I+\tau A)^{-k} (A^{-1} R^{n-k+1})
  \end{aligned}
\end{equation}
Suppose that
\begin{equation}
  \begin{aligned}
    |R^{n}| &\le |R| \\
    &:= C h^{\min\{r\alpha/2, 2\}}(x_i^{-\alpha} + (2T-x_i)^{-\alpha}) \\
    & \quad+ C(r-1) h^2 (T - \delta(x_i) + h_N)^{1-\alpha} + C \tau^{?}
  \end{aligned}
\end{equation}
Since $0 < A^{-1} R \le C h^{\min}$,
% Depose it by the basis $v_i$, we have
\begin{equation}
  \begin{aligned}
    |(I+\tau A)^{-k} \tau R^{n-k+1}| 
    \le (I+\tau A)^{-k} \tau R 
    = \tau A (1+\tau A)^{-k} A^{-1} R
  \end{aligned}
\end{equation}
Then
\begin{equation}
  \begin{aligned}
    |E^{n}| &\le |(I+\tau A)^{-n}E^{0}| + \sum_{k=1}^{n} \tau A (1+\tau A)^{-k} A^{-1} R  \\
    &= |(I+\tau A)^{-n}E^{0}| + \left(I - (I+\tau A)^{-n}\right) A^{-1} R.
  \end{aligned}
\end{equation}
Since $A$ is diagonally dominant, $\|(I+\tau A)^{-1} E\|_\infty \le \|E\|_\infty$, we have
\begin{equation}
  \|E^{n}\|_\infty \le \|E_{0}\|_\infty +  \|A^{-1} R\|_\infty.
\end{equation}

\begin{lemma}
  $A^{-1} R$ is bounded by $C \left(h^{\min\{r\alpha/2, 2\}} + \tau^{?}\right)$, where $C$ is a constant independent of $h, \alpha$.
\end{lemma}
